48  return false; //didn't read one
49  } else {

51  //make rect block

53  static_cast<int16_t>(x), static_cast<int16_t>(ysize - y - height),
54  static_cast<int16_t>(x + width), static_cast<int16_t>(ysize - y));
55  //on end of list
56  block_it.add_to_end (block);
57  }
58  fclose(pdfp);
59  }
