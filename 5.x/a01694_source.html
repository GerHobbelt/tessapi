24 inline std::string CodepointList(const std::vector<char32>& str32) {
25  std::stringstream result;
26  int total_chars = str32.size();
27  result << std::hex;
28  for (int i = 0; i < total_chars; ++i) {
29  result << "[" << str32[i] << "]";
30  }
31  return result.str();
32 }
